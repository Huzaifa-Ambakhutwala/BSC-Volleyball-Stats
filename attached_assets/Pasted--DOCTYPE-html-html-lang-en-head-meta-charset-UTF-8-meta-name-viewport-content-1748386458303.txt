                    <h3 class="text-2xl font-semibold text-slate-600">3.3 Stat Tracking Flow</h3>
                    <h4 class="text-xl font-medium text-slate-600 mt-3">Sets</h4>
                    <ul class="list-disc list-inside space-y-1 pl-4">
                        <li>Each match consists of up to 3 sets.</li>
                        <li>You can only enter stats for the current active set.</li>
                        <li>Once a set is finalized, it becomes locked, and you automatically move to the next set.</li>
                        <li>After the third set, the entire match can be submitted and finalized.</li>
                    </ul>

                    <h4 class="text-xl font-medium text-slate-600 mt-4">Actions (Stat Entry Types)</h4>
                    <p>Stat actions are intuitively grouped into three color-coded categories for easier understanding and faster tracking:</p>
                    
                    <div class="mt-4 p-4 border rounded-lg bg-slate-100">
                        <h5 class="text-lg font-semibold text-green-600 mb-2">‚úÖ Earned Actions (Green Buttons)</h5>
                        <p class="text-sm text-slate-600 mb-3">These represent positive plays that contribute points or successful attacks.</p>
                        <div class="stat-button-grid mb-4">
                            <div class="stat-button stat-button-green">Ace</div>
                            <div class="stat-button stat-button-green">Kill</div>
                            <div class="stat-button stat-button-green">Tip</div>
                            <div class="stat-button stat-button-green">Dump</div>
                            <div class="stat-button stat-button-green">Point</div>
                        </div>
                        <ul class="list-disc list-inside space-y-1 pl-4 text-sm">
                            <li><strong>Ace:</strong> A serve that lands untouched or cannot be returned by the opponent.</li>
                            <li><strong>Kill:</strong> A successful attack that lands and directly results in a point.</li>
                            <li><strong>Tip:</strong> A light, strategic touch over the net that earns a point.</li>
                            <li><strong>Dump:</strong> A strategic second-touch by the setter that results in a point.</li>
                            <li><strong>Point:</strong> A general point credited when needed (e.g., for an unknown opponent error).</li>
                        </ul>
                    </div>

                    <div class="mt-4 p-4 border rounded-lg bg-slate-100">
                        <h5 class="text-lg font-semibold text-blue-600 mb-2">üü¶ Block Actions (Blue/Teal Buttons)</h5>
                        <p class="text-sm text-slate-600 mb-3">These are defensive plays executed at the net.</p>
                        <div class="stat-button-grid mb-4">
                            <div class="stat-button stat-button-blue">Point Block</div>
                            <div class="stat-button stat-button-blue">Neutral Block</div>
                            <div class="stat-button stat-button-blue">Block (Point)</div>
                        </div>
                        <ul class="list-disc list-inside space-y-1 pl-4 text-sm">
                            <li><strong>Point Block:</strong> A block that directly results in a point for your team.</li>
                            <li><strong>Neutral Block:</strong> A block that deflects the ball but does not directly score a point.</li>
                            <li><strong>Block (Point):</strong> An alternate label for a point-scoring block.</li>
                        </ul>
                    </div>

                    <div class="mt-4 p-4 border rounded-lg bg-slate-100">
                        <h5 class="text-lg font-semibold text-red-600 mb-2">‚ùå Fault Actions (Red Buttons)</h5>
                        <p class="text-sm text-slate-600 mb-3">These indicate errors or violations that result in a point for the opposing team.</p>
                        <div class="stat-button-grid mb-4">
                            <div class="stat-button stat-button-red">Serve Error</div>
                            <div class="stat-button stat-button-red">Spike Error</div>
                            <div class="stat-button stat-button-red">Net Touch</div>
                            <div class="stat-button stat-button-red">Foot Fault</div>
                            <div class="stat-button stat-button-red">Reach</div>
                            <div class="stat-button stat-button-red">Carry</div>
                            <div class="stat-button stat-button-red">Out of Bounds</div>
                            <div class="stat-button stat-button-red">Generic Fault</div>
                        </div>
                        <ul class="list-disc list-inside space-y-1 pl-4 text-sm">
                            <li><strong>Serve Error:</strong> A missed or illegal serve.</li>
                            <li><strong>Spike Error:</strong> A failed attack (e.g., ball goes out of bounds or into the net).</li>
                            <li><strong>Net Touch:</strong> A player illegally touches the net during play.</li>
                            <li><strong>Foot Fault:</strong> A player steps over the line during a serve or other play.</li>
                            <li><strong>Reach:</strong> An illegal reach over or under the net.</li>
                            <li><strong>Carry:</strong> Prolonged contact with the ball (e.g., a lift or scoop).</li>
                            <li><strong>Out of Bounds:</strong> The ball lands out of bounds on your team‚Äôs hit.</li>
                            <li><strong>Generic Fault:</strong> For other violations not specifically listed above.</li>
                        </ul>

                    <p class="text-lg leading-relaxed">Understanding the terminology used within the Volleyball Stat Tracker is key to its effective use. This glossary defines common terms you'll encounter while navigating the platform and recording statistics.</p>
                    <div class="space-y-3 mt-4">
                        <p><strong class="text-slate-700">Set:</strong> One segment of a volleyball match. Matches are typically played as best of 3 or best of 5 sets.</p>
                        <p><strong class="text-slate-700">Stat Tracker:</strong> The designated team member or user responsible for entering live statistics during a match using the tracker interface.</p>
                        <p><strong class="text-slate-700">Admin:</strong> A user with full access and privileges for managing the site, including player/team setup, scheduling, password management, and system oversight.</p>
                        <p><strong class="text-slate-700">Action:</strong> A specific stat entry recorded during a match, such as a kill, ace, serve error, block, etc. These are categorized for clarity (Earned, Block, Fault).</p>
                        <p><strong class="text-slate-700">Kill:</strong> A successful attack by a player that results directly in a point.</p>
                        <p><strong class="text-slate-700">Ace:</strong> A serve that results directly in a point, either by landing in the opponent's court untouched or being mishandled by the receiving team.</p>
                        <p><strong class="text-slate-700">Block (Point):</strong> A defensive play at the net where a player successfully blocks an opponent's attack, resulting directly in a point.</p>
                        <p><strong class="text-slate-700">Error/Fault:</strong> An action by a player or team that results in the opposing team being awarded a point (e.g., serve error, spike error, net touch).</p>

        document.addEventListener('DOMContentLoaded', () => {
            const navLinks = document.querySelectorAll('.nav-link');
            const internalNavLinks = document.querySelectorAll('.nav-internal');
            const contentSections = document.querySelectorAll('.content-section');
            const sidebar = document.getElementById('sidebar');
            const mobileMenuButton = document.getElementById('mobileMenuButton');
            const mainContentArea = document.querySelector('main');

            function updateContent(hash) {
                contentSections.forEach(section => {
                    section.classList.remove('active');
                });
                navLinks.forEach(link => {
                    link.classList.remove('active');
                });

                const activeSection = document.querySelector(hash);
                const activeLink = document.querySelector(`.nav-link[href="${hash}"]`);
                
                if (activeSection) {
                    activeSection.classList.add('active');
                    // Scroll to top of content area
                    mainContentArea.scrollTop = 0;
                } else if (hash === '' || hash === '#') {
                    // Default to overview if hash is empty or just #
                    document.querySelector('#overview').classList.add('active');
                    document.querySelector('.nav-link[href="#overview"]').classList.add('active');
                }


                if (activeLink) {
                    activeLink.classList.add('active');
                    // If it's a sub-item, also activate its parent
                    const parentDiv = activeLink.closest('div.ml-4');
                    if (parentDiv) {
                        const parentLink = parentDiv.previousElementSibling;
                        if (parentLink && parentLink.classList.contains('nav-link')) {
                             parentLink.classList.add('active');
                        }
                    }
                }
                 // Close mobile menu on navigation
                if (window.innerWidth < 768 && sidebar.classList.contains('open')) {
                    sidebar.classList.remove('open');
                }
            }

            function handleNavigation(event) {
                event.preventDefault();
                const hash = event.currentTarget.hash;
                window.location.hash = hash; // Update URL hash
            }
            
            navLinks.forEach(link => {
                link.addEventListener('click', handleNavigation);
            });

            internalNavLinks.forEach(link => {
                 link.addEventListener('click', handleNavigation);
            });

            mobileMenuButton.addEventListener('click', () => {
                sidebar.classList.toggle('open');
            });
            
            // Close sidebar if clicking outside on mobile
            document.addEventListener('click', (event) => {
                if (window.innerWidth < 768 && sidebar.classList.contains('open')) {
                    if (!sidebar.contains(event.target) && !mobileMenuButton.contains(event.target)) {
                        sidebar.classList.remove('open');
                    }
                }
            });


            // Handle hash changes (e.g. browser back/forward buttons, direct link)
            window.addEventListener('hashchange', () => {
                updateContent(window.location.hash || '#overview');
            });

            // Initial content load
            updateContent(window.location.hash || '#overview');
        });
